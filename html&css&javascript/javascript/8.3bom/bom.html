会话数据/持久数据的存储、读取与清除
        
sessionStorage.setItem("keyA", "value_A") // 关闭会话后再打开就没了
localStorage.setItem("keyB", "value_B") // 关闭会话后再打开还在
console.log(sessionStorage.getItem("keyA")) // 读取会话数据并打印到控制台
console.log(localStorage.getItem("keyB")) // 读取存储数据并打印到控制台
sessionStorage.removeItem("keyA") // 同上不解释
localStorage.removeItem("keyB") // 同上不解释
